## 1. Dependency Installation

Run this single command from the project root:

```bash
npm install drizzle-orm @neondatabase/serverless next-auth@5.0.0-beta.25 @auth/drizzle-adapter @anthropic-ai/sdk ai @ai-sdk/anthropic zod bcryptjs uploadthing @uploadthing/react recharts react-hook-form @hookform/resolvers resend lucide-react clsx tailwind-merge class-variance-authority date-fns
```

```bash
npm install -D drizzle-kit @types/bcryptjs
```

Then initialize shadcn/ui (select "New York" style, "Zinc" base color, CSS variables = yes):

```bash
npx shadcn@latest init
```

Then add all required shadcn components:

```bash
npx shadcn@latest add button card input label select textarea dialog sheet tabs table badge separator avatar dropdown-menu form toast sonner skeleton switch checkbox radio-group scroll-area popover command alert alert-dialog tooltip progress
```

```typescript
export const authConfig: NextAuthConfig = {
  adapter: DrizzleAdapter(db),
  session: { strategy: "jwt" },
  pages: {
    signIn: "/login",
  },
  providers: [
    Credentials({
      name: "credentials",
      credentials: {
        email: { label: "Email", type: "email" },
        password: { label: "Password", type: "password" },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) return null;

        const user = await db.query.users.findFirst({
          where: eq(users.email, credentials.email as string),
        });
        if (!user) return null;

        const passwordValid = await bcrypt.compare(
          credentials.password as string,
          user.passwordHash
        );
        if (!passwordValid) return null;

        return {
          id: user.id,
          email: user.email,
          name: `${user.firstName} ${user.lastName}`,
          role: user.role,
        };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id;
        token.role = (user as any).role;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        session.user.id = token.id as string;
        (session.user as any).role = token.role as string;
      }
      return session;
    },
    async authorized({ auth, request: { nextUrl } }) {
      const isLoggedIn = !!auth?.user;
      const isOnPlatform = nextUrl.pathname.startsWith("/(platform)") ||
        (!nextUrl.pathname.startsWith("/login") &&
         !nextUrl.pathname.startsWith("/register") &&
         !nextUrl.pathname.startsWith("/forgot-password") &&
         !nextUrl.pathname.startsWith("/api/auth") &&
         nextUrl.pathname !== "/");

      if (isOnPlatform) {
        if (isLoggedIn) return true;
        return false; // redirects to signIn page
      }
      return true;
    },
  },
};
```

### 6.4 `lib/services/ai.service.ts` -- Core AI Logic

```typescript
import Anthropic from "@anthropic-ai/sdk";
import { db } from "@/lib/db";
import { exercises } from "@/lib/db/schema";
import { and, eq, inArray } from "drizzle-orm";
import { buildSystemPrompt, buildUserPrompt } from "@/lib/ai/prompts/workout-generation";
import { selectExerciseTool, selectExerciseSchema } from "@/lib/ai/tools/exercise-lookup";
import type { GeneratedExercise, GeneratedWorkout } from "@/lib/ai/schemas/workout-output";

const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY! });

interface GenerateWorkoutParams {
  patientId: string;
  limitations: string | null;
  comorbidities: string | null;
  functionalChallenges: string | null;
  equipment: string[];
  goals: string[];
  durationMinutes: number;
  daysPerWeek: number;
}

export async function generateWorkout(params: GenerateWorkoutParams): Promise<GeneratedWorkout> {
  // 1. Fetch candidate exercises filtered by patient equipment and active status
  const candidateExercises = await db.query.exercises.findMany({
    where: eq(exercises.isActive, true),
  });

  // 2. Pre-filter: remove exercises contraindicated for patient conditions
  const patientConditions = [
    params.limitations,
    params.comorbidities,
    params.functionalChallenges,
  ].filter(Boolean).join(" ").toLowerCase();

  const safeExercises = candidateExercises.filter((ex) => {
    if (!ex.contraindications || ex.contraindications.length === 0) return true;
    return !ex.contraindications.some((c) =>
      patientConditions.includes(c.toLowerCase())
    );
  });

  // 3. Build prompts
  const systemPrompt = buildSystemPrompt(safeExercises);
  const userPrompt = buildUserPrompt(params);

  // 4. Call Claude with tool_use
  const candidateMap = new Map(safeExercises.map((e) => [e.id, e]));
  const selectedExercises: GeneratedExercise[] = [];
  let orderIndex = 0;

  const response = await anthropic.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 4096,
    system: systemPrompt,
    tools: [selectExerciseTool],
    messages: [{ role: "user", content: userPrompt }],
  });

  // 5. Process tool use results
  // Claude may return multiple tool_use blocks in a single response,
  // or we may need to continue the conversation for multi-turn tool use
  let currentResponse = response;
  const conversationMessages: Anthropic.MessageParam[] = [
    { role: "user", content: userPrompt },
  ];

  while (currentResponse.stop_reason === "tool_use") {
    const assistantContent = currentResponse.content;
    conversationMessages.push({ role: "assistant", content: assistantContent });

    const toolResults: Anthropic.ToolResultBlockParam[] = [];

    for (const block of assistantContent) {
      if (block.type === "tool_use" && block.name === "select_exercise") {
        const input = block.input as any;
        const parsed = selectExerciseSchema.safeParse(input);

        if (!parsed.success) {
          toolResults.push({
            type: "tool_result",
            tool_use_id: block.id,
            content: `Invalid parameters: ${parsed.error.message}. Please try again with valid values.`,
            is_error: true,
          });
          continue;
        }

        const exercise = candidateMap.get(parsed.data.exercise_id);
        if (!exercise) {
          toolResults.push({
            type: "tool_result",
            tool_use_id: block.id,
            content: `Exercise ID ${parsed.data.exercise_id} not found in the library. Choose from the provided exercises only.`,
            is_error: true,
          });
          continue;
        }

        selectedExercises.push({
          exerciseId: exercise.id,
          exerciseName: exercise.name,
          sets: parsed.data.sets,
          reps: parsed.data.reps,
          durationSeconds: parsed.data.duration_seconds,
          restSeconds: parsed.data.rest_seconds,
          rationale: parsed.data.rationale,
          orderIndex: orderIndex++,
        });

        toolResults.push({
          type: "tool_result",
          tool_use_id: block.id,
          content: `Successfully added "${exercise.name}" to the plan at position ${orderIndex}.`,
        });
      }
    }

    conversationMessages.push({ role: "user", content: toolResults });

    currentResponse = await anthropic.messages.create({
      model: "claude-sonnet-4-20250514",
      max_tokens: 4096,
      system: systemPrompt,
      tools: [selectExerciseTool],
      messages: conversationMessages,
    });
  }

  // 6. Construct the final workout object
  return {
    title: `${params.goals.join(" & ")} Workout`,
    description: `AI-generated ${params.durationMinutes}-min workout targeting ${params.goals.join(", ")}`,
    exercises: selectedExercises,
    totalEstimatedMinutes: params.durationMinutes,
    warmUpExerciseCount: Math.min(2, selectedExercises.length),
    coolDownExerciseCount: Math.min(2, Math.max(0, selectedExercises.length - 2)),
  };
}
```

### 6.5 `app/api/ai/generate-workout/route.ts` -- Streaming Endpoint

```typescript
import { auth } from "@/lib/auth/auth";
import { generateWorkout } from "@/lib/services/ai.service";
import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";

const requestSchema = z.object({
  patientId: z.string().uuid(),
  limitations: z.string().nullable(),
  comorbidities: z.string().nullable(),
  functionalChallenges: z.string().nullable(),
  equipment: z.array(z.string()),
  goals: z.array(z.string()),
  durationMinutes: z.number().int().min(10).max(90),
  daysPerWeek: z.number().int().min(1).max(7),
});

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session?.user) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const body = await req.json();
  const parsed = requestSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json(
      { error: "Invalid input", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  try {
    const workout = await generateWorkout(parsed.data);
    return NextResponse.json({ success: true, data: workout });
  } catch (error) {
    console.error("AI generation error:", error);
    return NextResponse.json(
      { error: "Failed to generate workout. Please try again." },
      { status: 500 }
    );
  }
}
```

```typescript
    firstName: "Sarah",
    lastName: "Chen",
  }).returning();

  // 2. Create demo patient
  const [patient] = await db.insert(schema.users).values({
    email: "patient@demo.com",
    passwordHash,
    role: "patient",
    firstName: "James",
    lastName: "Wilson",
    dateOfBirth: "1965-03-15",
  }).returning();

  // 3. Create patient profile
  await db.insert(schema.patientProfiles).values({
    userId: patient.id,
    limitations: "Mild lower back pain, right knee stiffness",
    comorbidities: "Type 2 diabetes, controlled hypertension",
    functionalChallenges: "Difficulty with balance, gets up slowly from chairs",
    availableEquipment: ["chair", "resistance_band"],
    fitnessGoals: ["balance", "lower_body_strength", "mobility"],
    preferredDurationMinutes: 25,
    preferredDaysPerWeek: 3,
  });

  // 4. Link patient to clinician
  await db.insert(schema.patientClinicianLinks).values({
    patientId: patient.id,
    clinicianId: clinician.id,
    status: "active",
  });

  // 5. Insert exercises
  const exerciseIdMap = new Map<string, string>(); // name -> inserted id

  for (const ex of exercisesData) {
    const [inserted] = await db.insert(schema.exercises).values({
      name: ex.name,
      description: ex.description,
      bodyRegion: ex.bodyRegion as any,
      equipmentRequired: ex.equipmentRequired,
      difficultyLevel: ex.difficultyLevel as any,
      contraindications: ex.contraindications,
      instructions: ex.instructions,
      isActive: true,
      createdBy: clinician.id,
    }).returning();
    exerciseIdMap.set(ex.name, inserted.id);
  }

  // 6. Insert progressions (second pass after all exercises exist)
  for (const ex of exercisesData) {
    const exerciseId = exerciseIdMap.get(ex.name);
    if (!exerciseId) continue;

    const allLinks = [
      ...(ex.progressions || []),
      ...(ex.regressions || []),
    ];

    for (const link of allLinks) {
      const nextId = exerciseIdMap.get(link.nextExerciseName);
      if (!nextId) {
        console.warn(`Progression target "${link.nextExerciseName}" not found for "${ex.name}"`);
        continue;
      }
      await db.insert(schema.exerciseProgressions).values({
        exerciseId,
        nextExerciseId: nextId,
        direction: link.direction as any,
        orderIndex: link.orderIndex,
      });
    }
  }

  console.log(`Seeded ${exercisesData.length} exercises`);
  console.log("Seeding complete.");
}

seed().catch(console.error);
```

```typescript
export function formatDate(date: Date | string): string {
  const d = typeof date === "string" ? new Date(date) : date;
  return format(d, "MMM d, yyyy");
}

export function formatDateTime(date: Date | string): string {
  const d = typeof date === "string" ? new Date(date) : date;
  return format(d, "MMM d, yyyy 'at' h:mm a");
}

export function formatRelative(date: Date | string): string {
  const d = typeof date === "string" ? new Date(date) : date;
  if (isToday(d)) return `Today at ${format(d, "h:mm a")}`;
  if (isYesterday(d)) return `Yesterday at ${format(d, "h:mm a")}`;
  return formatDistanceToNow(d, { addSuffix: true });
}
```

## 14. Critical Implementation Details

### 14.1 NextAuth v5 + Credentials Provider Gotchas

- NextAuth v5 with credentials provider does NOT persist sessions to the database. It uses JWT only. The DrizzleAdapter is technically only needed for OAuth providers. However, including it does no harm and enables future OAuth support.
- The `authorize` callback in the credentials provider MUST return an object with `id`, `email`, `name`, and `role`. Any additional fields must be threaded through via `jwt` and `session` callbacks.
- The `authorized` callback in the config is what the middleware uses. Return `true` to allow, `false` to redirect to sign-in, or a `Response`/`NextResponse` for custom redirects.
- `middleware.ts` MUST be at the project root (not inside `app/` or `lib/`).

### 14.2 Drizzle + Neon Setup

- Use `@neondatabase/serverless` (the HTTP driver), NOT `pg` or `postgres`. The HTTP driver works in Vercel's edge and serverless runtimes without TCP connection issues.
- Import `neon` from `@neondatabase/serverless` and `drizzle` from `drizzle-orm/neon-http`.
- Drizzle's `query` API (relational queries like `db.query.users.findFirst`) requires passing the `schema` object to the `drizzle()` constructor. Without this, only the raw `db.select().from()` API works.
- GIN indexes on array columns (equipment_required, contraindications) require raw SQL in migrations. Drizzle-kit may not generate these automatically. After initial migration, add a manual SQL migration:
  ```sql
  CREATE INDEX exercises_equipment_gin ON exercises USING GIN (equipment_required);
  CREATE INDEX exercises_contraindications_gin ON exercises USING GIN (contraindications);
  ```

### 14.3 shadcn/ui + Tailwind v4

- shadcn/ui with Tailwind v4 uses CSS variables defined in `globals.css` via `@theme` blocks. After running `npx shadcn@latest init`, the init script will rewrite `globals.css` with the full shadcn color system.
- Do NOT fight the shadcn init. Let it replace the existing `globals.css` content.
- When adding shadcn components, they are generated into `components/ui/`. This path is configured in `components.json`.

### 14.4 Claude Tool Use Implementation

- The tool_use flow is multi-turn. Claude responds with `stop_reason: "tool_use"` when it wants to call a tool. You must process the tool calls, return `tool_result` blocks, and call the API again.
- Each `tool_use` block has a unique `id` that MUST be referenced in the corresponding `tool_result`.
- Set `is_error: true` on the `tool_result` if validation fails. Claude will attempt an alternative.
- The conversation `messages` array must alternate between `user` and `assistant` roles. Tool results are sent as the `user` role with `content` being an array of `tool_result` blocks.

### 14.5 HIPAA Considerations

- NEVER send identifiable patient data (name, email, DOB) to the Claude API. The AI service should strip PII before constructing prompts. Send only: limitations, comorbidities, functional challenges, equipment, goals. These are clinical descriptors, not identifiers.
- NEVER log patient health information to `console.log` in production. Use structured logging that excludes PHI fields.
- Set `HttpOnly`, `Secure`, and `SameSite=Strict` on session cookies. NextAuth handles this by default in production.
- Do NOT use `localStorage` or `sessionStorage` for any health data. Only the JWT session token should persist client-side.
- Add a `medical-disclaimer` component to every page that displays AI-generated content.

### 14.6 Authorization Enforcement Pattern

Every server action MUST follow this pattern:

```typescript
"use server";

import { auth } from "@/lib/auth/auth";

export async function someAction(input: SomeInput): Promise<ActionResponse<SomeOutput>> {
  // 1. Authentication check
  const session = await auth();
  if (!session?.user) {
    return { success: false, error: "Unauthorized" };
  }

  // 2. Role check (if needed)
  if (session.user.role !== "clinician") {
    return { success: false, error: "Forbidden" };
  }

  // 3. Input validation
  const parsed = someSchema.safeParse(input);
  if (!parsed.success) {
    return { success: false, error: parsed.error.issues[0].message };
  }

  // 4. Ownership/link check (if needed)
  // e.g., verify clinician is linked to this patient

  // 5. Business logic via service layer
  try {
    const result = await someService.doThing(parsed.data);
    return { success: true, data: result };
  } catch (error) {
    console.error("someAction error:", error);
    return { success: false, error: "An unexpected error occurred" };
  }
}
```

### 14.7 Vercel AI SDK Streaming Pattern

For the workout generation page, the recommended client-side pattern:

```typescript

  // Display partial exercises as they arrive

```

However, since we need the tool_use validation layer (exercises must exist in the library), the actual implementation should use a two-phase approach:
1. Client calls the API route
2. API route runs the full tool_use loop server-side (with validation)
3. Returns the complete validated result as JSON
4. For UX, show an indeterminate progress indicator during generation
